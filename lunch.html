How Full Auto Works
A Firebase Cloud Function runs on a schedule you set (e.g. every 25th of the month). It reads unpaid members from your Firebase DB, builds personalised messages, and sends them via the WhatsApp Business API — completely hands-free.
🔥
Firebase Functions
Scheduled Node.js function on your existing project
💬
WhatsApp Business API
Via Africa's Talking or Twilio (~Ksh 2–5 per message)

Setup Guide
1
Sign up for Africa's Talking WhatsApp sandbox
Go to africastalking.com → create account → WhatsApp → request sandbox access. You'll get an API Key and a sender number.
2
Install Firebase CLI & init Functions
npm install -g firebase-tools
firebase login
firebase init functions   # choose JavaScript, your existing project
3
Paste this into functions/index.js
const functions = require("firebase-functions");
const admin = require("firebase-admin");
const https = require("https");
admin.initializeApp();

// ── CONFIG ── set these via: firebase functions:config:set
// firebase functions:config:set wa.apikey="YOUR_AT_KEY" wa.sender="+254XXXXXXXXX" wa.day="25"

const WA_APIKEY = functions.config()?.wa?.apikey || "";
const WA_SENDER = functions.config()?.wa?.sender || "";
const SEND_DAY  = parseInt(functions.config()?.wa?.day || "25");
const STD_AMT   = parseInt(functions.config()?.wa?.std  || "420");

// Runs at 8 AM Nairobi time on the day configured above
exports.sendLunchReminders = functions.pubsub
  .schedule(`0 8 ${SEND_DAY} * *`)
  .timeZone("Africa/Nairobi")
  .onRun(async () => {
    const now    = new Date();
    const month  = now.toLocaleString("en-KE", { month: "long" });
    const year   = now.getFullYear();
    const period = `${month}_${year}`;
    const db     = admin.database();
    const snap   = await db.ref(`lunch/${period}/members`).once("value");
    const data   = snap.val() || {};
    const log    = {};

    for (const [id, m] of Object.entries(data)) {
      if (!m.phone || m.amount >= STD_AMT) continue;
      const balance = STD_AMT - m.amount;
      const msg = `Dear ${m.name}, your lunch contribution of Ksh ${STD_AMT} for ${month} ${year} is outstanding. Paid: Ksh ${m.amount}. Balance: Ksh ${balance}. Please pay promptly. Thank you.`;
      const result = await sendWA(m.phone, msg);
      log[id] = { sentAt: now.toISOString(), status: result, phone: m.phone };
    }

    await db.ref(`lunch/${period}/reminderLog`).set(log);
    console.log("Reminders sent:", log);
  });

function sendWA(phone, message) {
  return new Promise(resolve => {
    const body = JSON.stringify({
      username: "sandbox",          // change to your AT username in production
      to: phone, from: WA_SENDER,
      message, channel: "whatsapp"
    });
    const req = https.request({
      hostname: "api.africastalking.com",
      path: "/version1/messaging", method: "POST",
      headers: {
        "apiKey": WA_APIKEY,
        "Content-Type": "application/json",
        "Accept": "application/json"
      }
    }, res => {
      let d = "";
      res.on("data", c => d += c);
      res.on("end",  () => resolve(res.statusCode === 201 ? "sent" : "failed"));
    });
    req.on("error", () => resolve("error"));
    req.write(body); req.end();
  });
}
4
Set your secrets & deploy
firebase functions:config:set \
  wa.apikey="YOUR_AFRICA_TALKING_KEY" \
  wa.sender="+254XXXXXXXXX" \
  wa.day="25" \
  wa.std="420"

firebase deploy --only functions
The function will auto-fire at 8 AM Nairobi time on day 25 every month. Change wa.day to any day you prefer.
✓
That's it — reminders run automatically every month
Results are written to lunch/{period}/reminderLog in your Firebase DB so you can audit who was sent a message and when.
